{{/* Image Component Downloads an image from a url and creates variants based on the formats value. Falls back to the remote image if resource is unavailable Params ====== - class: Added to class - url: Remote url to download - loading: loading attribute value - filename: base filename without extension - formats: Slice containing formats to compress to - width: image width - height: image height - alt: image alt text */}} {{ $original := "" }} {{/* Download Image */}} {{- with resources.GetRemote .url -}} {{ $src := . | resources.Copy (print $.filename "." .MediaType.SubType) }} {{- with .Err -}} {{/* Log Error if 404 */}} {{- warnf "%v" . -}} {{- else -}} {{ $original = $src.RelPermalink }} {{ range $.formats }} {{ with $src.Resize .options | resources.Copy (print $.filename .extension) }} {{ with .Err }} {{ warnf "%v" .Err }} {{ else }} {{ end }} {{ end }} {{ end }} {{/* Original Image */}} {{- end -}} {{- end -}} {{/* Fallback to Remote Image + sizes */}}
